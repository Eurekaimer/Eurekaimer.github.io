<script>
  // 1. 定义友链数据
  // 在 relations 中填入对方的 name 即可建立次级连线
  const friendsData = [
    {
      name: "Eurekaimer",
      img: "https://avatars.githubusercontent.com/u/88918522?s=400&u=c9494d29a9573c1d393929f5ebb785f2613ff4e7&v=4",
      desc: "学术主页 - 也就是博主本人",
      link: "https://www.eurekaimer.xyz/Academic/",
      isCenter: true 
    },
    {
      name: "Meraki's bar",
      img: "https://avatars.githubusercontent.com/u/187371253?v=4",
      desc: "一位学弟的博客",
      link: "https://meraki111.netlify.app/",
      // Wcowin 和 Meraki 认识，只要在其中一人这边写上对方名字即可
      relations: ["Wcowin"] 
    },
    {
      name: "Ice Year",
      img: "https://blog.iceyear.eu.org/android-chrome-512x512.webp",
      desc: "Il n’y a pas de hasard, il n’y a que des rendez-vous.",
      link: "https://blog.iceyear.eu.org",
      // Ice Year 认识 Yon Zilch 和 春风少年兄
      relations: ["Yon Zilch", "春风少年兄"]
    },
    {
      name: "春风少年兄",
      img: "https://avatars.githubusercontent.com/u/138082074?v=4",
      desc: "你在世纪大道东门",
      link: "https://blog.0pt.icu/",
      // 春风少年兄 认识 Yon Zilch (Ice Year 已经在上面连过他了，这里不用重复连 Ice Year)
      relations: ["Yon Zilch"]
    },
    {
      name: "Yon Zilch",
      img: "https://avatars.githubusercontent.com/u/141223334?v=4",
      desc: "From the yon, into zilch. Life, a vanished dream.",
      link: "https://blog.yon.im/"
    },
    {
      name: "Wcowin",
      img: "https://pic4.zhimg.com/80/v2-a0456a5f527c1923f096759f2926012f_1440w.webp",
      desc: "循此苦旅，以达星辰",
      link: "https://wcowin.work/"
    }
  ];

  // --- 核心工具函数：将图片转换为圆形 Base64 (保持不变) ---
  const getCircleImage = (url) => {
    return new Promise((resolve) => {
      const img = new Image();
      img.crossOrigin = 'Anonymous'; 
      img.src = url;
      img.onload = () => {
        const canvas = document.createElement('canvas');
        const size = Math.min(img.width, img.height);
        canvas.width = size;
        canvas.height = size;
        const ctx = canvas.getContext('2d');
        ctx.beginPath();
        ctx.arc(size / 2, size / 2, size / 2, 0, Math.PI * 2);
        ctx.closePath();
        ctx.clip();
        ctx.drawImage(img, (img.width - size) / 2, (img.height - size) / 2, size, size, 0, 0, size, size);
        try { resolve('image://' + canvas.toDataURL()); } 
        catch (e) { resolve('image://' + url); }
      };
      img.onerror = () => { resolve('image://' + url); };
    });
  };

  // --- 主逻辑 ---
  const initGraph = async () => {
    const chartDom = document.getElementById('friend-graph');
    const myChart = echarts.init(chartDom);
    myChart.showLoading();

    // 1. 处理节点和图片
    const processedNodes = await Promise.all(friendsData.map(async (item) => {
        const circleImg = await getCircleImage(item.img);
        
        // 调整大小：你是 100， 朋友是 50
        const size = item.isCenter ? 100 : 50;
        
        return {
            name: item.name,
            symbol: circleImg,
            symbolSize: size,
            value: item.desc, 
            link: item.link,
            // 你的名字加粗且字号大一点
            label: {
                show: true,
                position: 'bottom',
                color: 'inherit',
                fontSize: item.isCenter ? 16 : 12,
                fontWeight: item.isCenter ? 'bold' : 'normal',
                distance: 10
            }
        };
    }));

    // 2. 构建连线
    const edges = [];
    const centerNode = friendsData.find(f => f.isCenter);
    
    // (A) 主线：所有人连向你
    processedNodes.forEach(node => {
        if (node.name !== centerNode.name) {
            edges.push({
                source: centerNode.name,
                target: node.name,
                lineStyle: {
                    color: '#999',
                    width: 2, // 主线稍微粗一点
                    opacity: 0.6,
                    curveness: 0.1
                }
            });
        }
    });

    // (B) 次级线：朋友互联
    friendsData.forEach(item => {
        if (item.relations && item.relations.length > 0) {
            item.relations.forEach(targetName => {
                edges.push({
                    source: item.name,
                    target: targetName,
                    lineStyle: {
                        color: '#608DBD', // 次级关系用稍微不同的颜色（淡蓝色）
                        width: 1.5,
                        opacity: 0.5,
                        curveness: 0.2 // 稍微弯一点，避免直线重叠
                    }
                });
            });
        }
    });

    // 3. 配置项
    const option = {
        backgroundColor: 'transparent',
        
        tooltip: {
            trigger: 'item',
            backgroundColor: 'rgba(50, 50, 50, 0.9)',
            borderColor: '#333',
            textStyle: { color: '#fff' },
            formatter: function (params) {
                if (params.dataType === 'node') {
                    return `
                        <div style="text-align:center; padding: 5px;">
                        <strong style="font-size:14px;">${params.name}</strong><br/>
                        <span style="font-size:12px; opacity: 0.8; line-height: 1.5;">${params.data.value}</span>
                        </div>
                    `;
                }
            }
        },

        series: [{
            type: 'graph',
            layout: 'force',
            data: processedNodes,
            links: edges,
            roam: true,
            draggable: true,
            // 调整力学参数，适应变大的节点
            force: {
                repulsion: 800, // 斥力调大，防止大节点重叠
                edgeLength: [100, 200], // 连线长度范围
                gravity: 0.08
            },
            itemStyle: {
                borderColor: '#fff',
                borderWidth: 2,
                shadowBlur: 10,
                shadowColor: 'rgba(0, 0, 0, 0.2)'
            },
            // 选中高亮样式
            emphasis: {
                focus: 'adjacency', // 鼠标悬停时，只高亮相关联的节点和边
                lineStyle: {
                    width: 4
                }
            }
        }]
    };

    myChart.hideLoading();
    myChart.setOption(option);

    myChart.on('click', function (params) {
        if (params.dataType === 'node' && params.data.link) {
            window.open(params.data.link, '_blank');
        }
    });

    window.addEventListener('resize', function () {
        myChart.resize();
    });
  };

  initGraph();
</script>
